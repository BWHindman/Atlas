#prepare
mkdir /projects/hindman/src
cd /projects/hindman/src

#extract
wget http://heasarc.gsfc.nasa.gov/FTP/software/fitsio/c/cfitsio3450.tar.gz
tar -xvf cfitsio3450.tar.gz
cd cfitsio

#configure and build
ml purge
ml intel
./configure --prefix=/projects/hindman/software/cfitsio/3.4.5 CFLAGS="-g -O2  -xCORE-AVX2 -ip -ipo"
make
make shared  <--- important
make install

#Build Atlas
cd <atlas directory>
(1) replace makefile and fitswrapper.c with versions I sent you
(1) Modify makefile I sent so that any instances of feathern are replaced by hindman
ml purge
ml intel impi mkl
make

# Run Atlas
I don't know how to do this, but before you can run it, you will absolutely need to run the following commands:
ml purge
ml intel impi mkl
export LD_LIBRARY_PATH=/projects/hindman/software/cfitsio/3.4.5/lib:$LD_LIBRARY_PATH

If you don't do the last one, you will get an error related to cfitsio not being found
